    Messaging Mixin

    Add functionality for showing messages related to a form field.

    mixin(Component, messaging);

     setError() 

    Set the error state.

    Option name Type Description
    message String

    Optional

    		 
    setError(message) {

  // Animate down
  if (!this._isMessageVisible()) {
    this._showMessage();
  }

  this.clearWarning();
  this.clearSuccess();

  this.el.setAttribute('data-error', true);

  if (message) {
    this.setMessage(message);
  }

  return this;
},

    _showMessage

    method
     _showMessage() 

    Show the message

    _showMessage() {

  if (!this.messageEl.parentNode) {
    this.el.appendChild(this.messageEl);
  }

  animateHeight({
    el: this.el,
    toggleEl: this.messageEl
  });
},

    _hideMessage

    method
     _hideMessage() 

    Hide the message.

    Option name Type Description
    callback Function 
    _hideMessage(callback) {

  animateHeight({
    el: this.el,
    toggleEl: this.messageEl,
    toggleValue: 'none',
    action: 'collapse',
    onComplete: callback
  });
},

    return Boolean 
    _isMessageVisible() {
  return this.el.getAttribute('data-error') || this.el.getAttribute('data-warning') || this.el.getAttribute('data-success');
}
